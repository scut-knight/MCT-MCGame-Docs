35 -(void)loadSoundConfiguration{
36  NSString * filepath = [self filePath];
37  if ([[NSFileManager defaultManager]fileExistsAtPath:filepath]) {
38  NSDictionary * configDic = [[NSDictionary alloc]initWithContentsOfFile:filepath];
39  [self set_BackGroundMusicSwitch: [configDic objectForKey:S_L_BackGroundMusicSwitch]];
40  [self set_RotateEffectSwitch: [configDic objectForKey:S_L_RotateEffectSwitch]];
41  [self set_BackGroundMusicValume:[configDic objectForKey:S_L_BackGroundMusicValume]];
42  [self set_RotateEffectValume:[configDic objectForKey:S_L_RotateEffectValume]];
43  [configDic release];
44 
45  }else{
46  [self set_BackGroundMusicValume:[[[NSNumber alloc] initWithFloat:1.0] autorelease]];
47  [self set_RotateEffectValume:[[[NSNumber alloc] initWithFloat:1.0] autorelease]];
48  [self set_RotateEffectSwitch :[[[NSNumber alloc] initWithBool:YES] autorelease]];
49  [self set_BackGroundMusicSwitch:[[[NSNumber alloc] initWithBool:YES] autorelease]];
50  }
51  if ([[self _BackGroundMusicSwitch]boolValue]) {
52  backgroundPlayer = [MCSoundBoard audioPlayerForKey:Audio_BackGroundSound_Loop_key];
53  backgroundPlayer.numberOfLoops = -1; // Endless

55  NSNumber *maxvolume = [[[NSNumber alloc]initWithFloat:backgroundPlayer.volume] autorelease];
56  [MCSoundBoard playAudioForKey:Audio_BackGroundSound_Loop_key fadeInInterval:2.0 maxVolume:maxvolume];
57  }
