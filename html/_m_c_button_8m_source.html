84 -(void)handleTouches
85 {
86  NSSet * touches = [[[CoordinatingController sharedCoordinatingController] currentController].inputController touchEvents];
87  if ([touches count] == 0) return;
88  BOOL pointInBounds = NO;
89  FSM_Interaction_State fsm_Current_State = [[[CoordinatingController sharedCoordinatingController] currentController].inputController fsm_Current_State];
90  CGPoint lastpoint = [[[CoordinatingController sharedCoordinatingController] currentController].inputController lastpoint];
91  // 处理每一点触控
92  for (UITouch * touch in [touches allObjects]) {
93  CGPoint touchPoint = [touch previousLocationInView:[touch view]];
94  if (touch.phase == UITouchPhaseEnded && fsm_Current_State==kState_F1){
95  if (CGRectContainsPoint(screenRect, lastpoint)) {
96  [self touchUp];
97  break;
98  }
99  }
100  if (CGRectContainsPoint(screenRect, touchPoint)) {
101  pointInBounds = YES;
102  if ((touch.phase == UITouchPhaseBegan)||((touch.phase == UITouchPhaseStationary))) {
103  [self touchDown];
104  }
105  }
106  }
107  if (!pointInBounds) {
108  [self showUp];
109  }
