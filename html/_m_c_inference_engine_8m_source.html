142 - (NSInteger)treeNodesApply:(MCTreeNode *)root withDeep:(NSInteger)deep{
143  // current MCMagicCube
144  NSObject<MCMagicCubeDataSouceDelegate> *mcDataSource = self.workingMemory.magicCube;
145 
146  switch (root.type) {
147  case ExpNode:
148  {
149  switch (root.value) {
150  case And:
151  {
152  root.result = [self andNodeApply:root withDeep:deep];
153  return root.result;
154  }
155  case Or:
156  {
157  root.result = [self orNodeApply:root withDeep:deep];
158  return root.result;
159  }
160  case Not:
161  {
162  root.result = [self notNodeApply:root withDeep:deep];
163  return root.result;
164  }
165  default:
166  break;
167  }
168  }
169  break;
170 
171  case PatternNode:
172  {
173 
174  switch (root.value) {
175  case Home:
176  {
177  ColorCombinationType value;
178  for (MCTreeNode *child in root.children) {
179  value = (ColorCombinationType)[self treeNodesApply:child withDeep:deep+1];
180  if (![self isCubieAtHomeWithIdentity:value]) {
181  root.result = NO;
182  return root.result;
183  }
184  }
185  }
186  break;
187 
188  case Check:
189  {
190  ColorCombinationType targetCubie = ColorCombinationTypeBound;
191  for (MCTreeNode *subPattern in root.children) {
192 
193  switch (subPattern.value) {
194  case At:
195  {
196  targetCubie = (ColorCombinationType)[self treeNodesApply:[subPattern.children objectAtIndex:0] withDeep:deep+1];
197  ColorCombinationType targetPosition = (ColorCombinationType)[self treeNodesApply:[subPattern.children objectAtIndex:1] withDeep:deep+1];
198  struct Point3i coorValue = [mcDataSource coordinateValueOfCubieWithColorCombination:(ColorCombinationType)targetCubie];
199  // if the position of targetCube is not the targetPosition
200  if (coorValue.x + coorValue.y*3 + coorValue.z * 9 + 13 != targetPosition) {
201  root.result = NO;
202  return root.result;
203  }
204  }
205  break;
206  case ColorBindOrientation:
207  {
208  NSObject<MCCubieDelegate> *cubie = nil;
209  // cubie will be the special cube in appropriate place
210  FaceOrientationType targetOrientation = (FaceOrientationType)[self treeNodesApply:[subPattern.children objectAtIndex:0] withDeep:deep+1];
211  FaceColorType targetColor = (FaceColorType)[self treeNodesApply:[subPattern.children objectAtIndex:1] withDeep:deep+1];
212  if ([subPattern.children count] > 2) {
213  NSInteger position = [(MCTreeNode *)[subPattern.children objectAtIndex:2] value];
214  cubie = [mcDataSource cubieAtCoordinateX:(position%3-1) Y:(position%9/3-1) Z:(position/9-1)];
215  } else {
216  cubie = [mcDataSource cubieWithColorCombination:(ColorCombinationType)targetCubie];
217  }
218  root.result = [cubie isFaceColor:targetColor inOrientation:targetOrientation];
219  return root.result;
220  }
221  case NotAt:
222  {
223  targetCubie = (ColorCombinationType)[self treeNodesApply:[subPattern.children objectAtIndex:0] withDeep:deep+1];
224  ColorCombinationType targetPosition = (ColorCombinationType)[self treeNodesApply:[subPattern.children objectAtIndex:1] withDeep:deep+1];
225  struct Point3i coorValue = [mcDataSource coordinateValueOfCubieWithColorCombination:(ColorCombinationType)targetCubie];
226  // if the position of targetCube is the targetPosition
227  if (coorValue.x + coorValue.y*3 + coorValue.z * 9 + 13 == targetPosition) {
228  root.result = NO;
229  return root.result;
230  }
231  }
232  break;
233  default:
234  {
235  root.result = NO;
236  return root.result;
237  }
238  }
239  }
240  }
241  break;
242 
243  case CubiedBeLocked:
244  {
245  int index = 0;
246  if ([root.children count] != 0) {
247  index = [(MCTreeNode *)[root.children objectAtIndex:0] value];
248  }
249 
250  if ([self.workingMemory lockerEmptyAtIndex:index]) {
251  root.result = NO;
252  return root.result;
253  }
254  }
255  break;
256  default:
257  {
258  root.result = NO;
259  return root.result;
260  }
261  }
262  }
263  break;
264 
265  case InformationNode:
266 
267  switch (root.value) {
268  case getCombinationFromOrientation:
269  {
270  int x=1, y=1, z=1;
271  for (MCTreeNode *child in root.children) {
272 
273  switch ([mcDataSource centerMagicCubeFaceInOrientation:(FaceOrientationType)child.value]) {
274  case Up:
275  y = 2;
276  break;
277  case Down:
278  y = 0;
279  break;
280  case Left:
281  x = 0;
282  break;
283  case Right:
284  x = 2;
285  break;
286  case Front:
287  z = 2;
288  break;
289  case Back:
290  z = 0;
291  break;
292  default:
293  break;
294  }
295  }
296 
297  //Store the result at the node
298  root.result = x+y*3+z*9;
299  return root.result;
300  }
301  case getCombinationFromColor:
302  {
303  int x=1, y=1, z=1;
304  for (MCTreeNode *child in root.children) {
305 
306  switch ((FaceColorType)[self treeNodesApply:child withDeep:deep+1]) {
307  case UpColor:
308  y = 2;
309  break;
310  case DownColor:
311  y = 0;
312  break;
313  case LeftColor:
314  x = 0;
315  break;
316  case RightColor:
317  x = 2;
318  break;
319  case FrontColor:
320  z = 2;
321  break;
322  case BackColor:
323  z = 0;
324  break;
325  default:
326  break;
327  }
328  }
329 
330  //Store the result at the node
331  root.result = x+y*3+z*9;
332  return root.result;
333  }
334  case getFaceColorFromOrientation:
335  {
336  FaceColorType color;
337  FaceOrientationType orientation = (FaceOrientationType)[(MCTreeNode *)[root.children objectAtIndex:0] value];
338  if ([root.children count] == 1) {
339 
340  switch ([mcDataSource centerMagicCubeFaceInOrientation:orientation]) {
341  case Up:
342  color = UpColor;
343  break;
344  case Down:
345  color = DownColor;
346  break;
347  case Left:
348  color = LeftColor;
349  break;
350  case Right:
351  color = RightColor;
352  break;
353  case Front:
354  color = FrontColor;
355  break;
356  case Back:
357  color = BackColor;
358  break;
359  default:
360  color = NoColor;
361  break;
362  }
363  }
364  else{
365  int value = [(MCTreeNode *)[root.children objectAtIndex:1] value];
366  struct Point3i coordinate = {value%3-1, value%9/3-1, value/9-1};
367  color = [[mcDataSource cubieAtCoordinatePoint3i:coordinate] faceColorInOrientation:orientation];
368  }
369 
370  //Store the result at the node
371  root.result = color;
372  return root.result;
373  }
374  case lockedCubie:
375  {
376  int index = 0;
377  if ([root.children count] != 0) {
378  index = [(MCTreeNode *)[root.children objectAtIndex:0] value];
379  }
380 
381  if ([self.workingMemory lockerEmptyAtIndex:index]) {
382  root.result = -1;
383  }
384  else{
385  root.result = [[self.workingMemory cubieLockedInLockerAtIndex:index] identity];
386  }
387 
388  return root.result;
389  }
390  default:
391  break;
392  }
393  break;
394 
395  case ElementNode:
396  return root.value;
397  default:
398  {
399  root.result = NO;
400  return root.result;
401  }
402  }
403 
404  root.result = YES;
405  return root.result;
406 }

447 - (NSString *)checkStateFromInit:(BOOL)isCheckStateFromInit;{
448  NSString *goStr;
449  //to check from init or not(是否需要初始化)
450  if (isCheckStateFromInit) {
451  goStr = START_STATE;
452  [self.workingMemory unlockAllCubies];
453  }
454  else{
455  goStr = _workingMemory.magicCubeState;
456  }
457  //check state
458  MCState *tmpState = [states objectForKey:goStr];
459  for (; tmpState != nil && [self treeNodesApply:[tmpState root] withDeep:0]; tmpState = [states objectForKey:goStr]) {
460  // 更新goStr
461  goStr = tmpState.afterState;
462  }
463  // if goStr != magicCubeState
464  if ([goStr compare:_workingMemory.magicCubeState] != NSOrderedSame) {
465  _workingMemory.magicCubeState = goStr;
466  [self.workingMemory unlockCubieAtIndex:0];
467  [self.workingMemory unlockCubiesInRange:NSMakeRange(4, CubieCouldBeLockMaxNum-4)];
468  [self reloadRulesAccordingToCurrentStateOfRubiksCube];
469  }
470 
471  return _workingMemory.magicCubeState;
472 }

482 - (MCRule *)reasoning{
483  [self checkStateFromInit:NO];
484 
485  NSLog(@"State:%@", _workingMemory.magicCubeState);
486 
487  // Check special rules firstly.
488  for (NSString *key in [self.specialRules allKeys])
489  {
490  if ([self applyPatternWihtPatternName:key ofType:Special]) {
491  //--------------------------
492  NSLog(@"Rules:%@", key);
493  self.workingMemory.agendaPattern = [self.specialPatterns objectForKey:key];
494  return [self.specialRules objectForKey:key];
495  }
496  }
497 
498  // If no special rules match the state, check general rules.
499  for (NSString *key in [self.rules allKeys])
500  {
501  if ([self applyPatternWihtPatternName:key ofType:General]) {
502  //--------------------------
503  NSLog(@"Rules:%@", key);
504  self.workingMemory.agendaPattern = [self.patterns objectForKey:key];
505  return [self.rules objectForKey:key];
506  }
507  }
508  return nil;
509 }
