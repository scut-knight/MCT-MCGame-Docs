186 int glhInvertMatrixf2(const float *m, float *out){
187  float wtmp[4][8];
188  float m0, m1, m2, m3, s;
189  float *r0, *r1, *r2, *r3;
190  r0 = wtmp[0], r1 = wtmp[1], r2 = wtmp[2], r3 = wtmp[3];
191  /*
192  wtmp 0 1 2 3
193  0-3 m[0][0-3] m[1][0-3] m[2][0-3] m[3][0-3]
194  4 1.0 0.0 0.0 0.0
195  5 0.0 1.0 0.0 0.0
196  6 0.0 0.0 1.0 0.0
197  7 0.0 0.0 0.0 1.0
198  */
199  r0[0] = MAT(m, 0, 0), r0[1] = MAT(m, 0, 1),
200  r0[2] = MAT(m, 0, 2), r0[3] = MAT(m, 0, 3),
201  r0[4] = 1.0, r0[5] = r0[6] = r0[7] = 0.0,
202 
203  r1[0] = MAT(m, 1, 0), r1[1] = MAT(m, 1, 1),
204  r1[2] = MAT(m, 1, 2), r1[3] = MAT(m, 1, 3),
205  r1[5] = 1.0, r1[4] = r1[6] = r1[7] = 0.0,
206 
207  r2[0] = MAT(m, 2, 0), r2[1] = MAT(m, 2, 1),
208  r2[2] = MAT(m, 2, 2), r2[3] = MAT(m, 2, 3),
209  r2[6] = 1.0, r2[4] = r2[5] = r2[7] = 0.0,
210 
211  r3[0] = MAT(m, 3, 0), r3[1] = MAT(m, 3, 1),
212  r3[2] = MAT(m, 3, 2), r3[3] = MAT(m, 3, 3),
213  r3[7] = 1.0, r3[4] = r3[5] = r3[6] = 0.0;
214 
215  // 通过初等列变换来实现单位矩阵的转换
216  // 查找第一行的最大值，并进行列的交换，以方便化简出单位矩阵。如果第一列全为0，则不存在矩阵的逆
217  /* choose pivot - or die */
218  if (fabsf(r3[0]) > fabsf(r2[0]))
219  SWAP_ROWS_FLOAT(r3, r2);
220  if (fabsf(r2[0]) > fabsf(r1[0]))
221  SWAP_ROWS_FLOAT(r2, r1);
222  if (fabsf(r1[0]) > fabsf(r0[0]))
223  SWAP_ROWS_FLOAT(r1, r0);
224  if (0.0 == r0[0])
225  return 0;
226  /* eliminate first variable */
227  m1 = r1[0] / r0[0];
228  m2 = r2[0] / r0[0];
229  m3 = r3[0] / r0[0];
230  s = r0[1]; //比例因子
231  r1[1] -= m1 * s;
232  r2[1] -= m2 * s;
233  r3[1] -= m3 * s;
234  s = r0[2];
235  r1[2] -= m1 * s;
236  r2[2] -= m2 * s;
237  r3[2] -= m3 * s;
238  s = r0[3];
239  r1[3] -= m1 * s;
240  r2[3] -= m2 * s;
241  r3[3] -= m3 * s;
242  s = r0[4];
243  if (s != 0.0) {
244  r1[4] -= m1 * s;
245  r2[4] -= m2 * s;
246  r3[4] -= m3 * s;
247  }
248  s = r0[5];
249  if (s != 0.0) {
250  r1[5] -= m1 * s;
251  r2[5] -= m2 * s;
252  r3[5] -= m3 * s;
253  }
254  s = r0[6];
255  if (s != 0.0) {
256  r1[6] -= m1 * s;
257  r2[6] -= m2 * s;
258  r3[6] -= m3 * s;
259  }
260  s = r0[7];
261  if (s != 0.0) {
262  r1[7] -= m1 * s;
263  r2[7] -= m2 * s;
264  r3[7] -= m3 * s;
265  }
266  // 以下同理
267  /* choose pivot - or die */
268  if (fabsf(r3[1]) > fabsf(r2[1]))
269  SWAP_ROWS_FLOAT(r3, r2);
270  if (fabsf(r2[1]) > fabsf(r1[1]))
271  SWAP_ROWS_FLOAT(r2, r1);
272  if (0.0 == r1[1])
273  return 0;
274  /* eliminate second variable */
275  m2 = r2[1] / r1[1];
276  m3 = r3[1] / r1[1];
277  r2[2] -= m2 * r1[2];
278  r3[2] -= m3 * r1[2];
279  r2[3] -= m2 * r1[3];
280  r3[3] -= m3 * r1[3];
281  s = r1[4];
282  if (0.0 != s) {
283  r2[4] -= m2 * s;
284  r3[4] -= m3 * s;
285  }
286  s = r1[5];
287  if (0.0 != s) {
288  r2[5] -= m2 * s;
289  r3[5] -= m3 * s;
290  }
291  s = r1[6];
292  if (0.0 != s) {
293  r2[6] -= m2 * s;
294  r3[6] -= m3 * s;
295  }
296  s = r1[7];
297  if (0.0 != s) {
298  r2[7] -= m2 * s;
299  r3[7] -= m3 * s;
300  }
301 
302  /* choose pivot - or die */
303  if (fabsf(r3[2]) > fabsf(r2[2]))
304  SWAP_ROWS_FLOAT(r3, r2);
305  if (0.0 == r2[2])
306  return 0;
307  /* eliminate third variable */
308  m3 = r3[2] / r2[2];
309  r3[3] -= m3 * r2[3], r3[4] -= m3 * r2[4],
310  r3[5] -= m3 * r2[5], r3[6] -= m3 * r2[6], r3[7] -= m3 * r2[7];
311  /* last check */
312  if (0.0 == r3[3])
313  return 0;
314 
315  s = 1.0 / r3[3]; /* now back substitute row 3 */
316  r3[4] *= s;
317  r3[5] *= s;
318  r3[6] *= s;
319  r3[7] *= s;
320 
321  m2 = r2[3]; /* now back substitute row 2 */
322  s = 1.0 / r2[2];
323  r2[4] = s * (r2[4] - r3[4] * m2), r2[5] = s * (r2[5] - r3[5] * m2),
324  r2[6] = s * (r2[6] - r3[6] * m2), r2[7] = s * (r2[7] - r3[7] * m2);
325  m1 = r1[3];
326  r1[4] -= r3[4] * m1, r1[5] -= r3[5] * m1,
327  r1[6] -= r3[6] * m1, r1[7] -= r3[7] * m1;
328  m0 = r0[3];
329  r0[4] -= r3[4] * m0, r0[5] -= r3[5] * m0,
330  r0[6] -= r3[6] * m0, r0[7] -= r3[7] * m0;
331 
332  m1 = r1[2]; /* now back substitute row 1 */
333  s = 1.0 / r1[1];
334  r1[4] = s * (r1[4] - r2[4] * m1), r1[5] = s * (r1[5] - r2[5] * m1),
335  r1[6] = s * (r1[6] - r2[6] * m1), r1[7] = s * (r1[7] - r2[7] * m1);
336  m0 = r0[2];
337  r0[4] -= r2[4] * m0, r0[5] -= r2[5] * m0,
338  r0[6] -= r2[6] * m0, r0[7] -= r2[7] * m0;
339 
340  m0 = r0[1]; /* now back substitute row 0 */
341  s = 1.0 / r0[0];
342  r0[4] = s * (r0[4] - r1[4] * m0), r0[5] = s * (r0[5] - r1[5] * m0),
343  r0[6] = s * (r0[6] - r1[6] * m0), r0[7] = s * (r0[7] - r1[7] * m0);
344 
345  // 下面为所求矩阵的逆
346  MAT(out, 0, 0) = r0[4];
347  MAT(out, 0, 1) = r0[5], MAT(out, 0, 2) = r0[6];
348  MAT(out, 0, 3) = r0[7], MAT(out, 1, 0) = r1[4];
349  MAT(out, 1, 1) = r1[5], MAT(out, 1, 2) = r1[6];
350  MAT(out, 1, 3) = r1[7], MAT(out, 2, 0) = r2[4];
351  MAT(out, 2, 1) = r2[5], MAT(out, 2, 2) = r2[6];
352  MAT(out, 2, 3) = r2[7], MAT(out, 3, 0) = r3[4];
353  MAT(out, 3, 1) = r3[5], MAT(out, 3, 2) = r3[6];
354  MAT(out, 3, 3) = r3[7];
355  return 1;
356 }
