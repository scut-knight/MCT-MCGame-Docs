54 -(void)awake
55 {
56  // alloc some space for our particles
57  if (activeParticles == nil) activeParticles = [[NSMutableArray alloc] init];
58 
59  // we are going to pre-alloc a whole bunch of particles so
60  // we dont wast time during gameplay allocing them
61  // 先初始化到粒子池中
62  particlePool = [[NSMutableArray alloc] initWithCapacity:MC_MAX_PARTICLES];
63  NSInteger count = 0;
64  for (count = 0; count < MC_MAX_PARTICLES; count++) {
65  MCParticle * p = [[MCParticle alloc] init];
66  [particlePool addObject:p];
67  [p release];
68  }
69 
70  // finally make some space for our final particle mesh
71  // our vertexes will be all 3 axes, and we need 6 axes per particle
72  // 为粒子系统所在图层的描述坐标的数组分配内存
73  vertexes = (CGFloat *) malloc(3 * 6 * MC_MAX_PARTICLES * sizeof(CGFloat));
74  uvCoordinates = (CGFloat *) malloc(2 * 6 * MC_MAX_PARTICLES * sizeof(CGFloat));
75 }

82 -(void)setParticle:(NSString*)atlasKey
83 {
84  // ok, here we are going to go and get the quad that will be our particle image
85  // however, we do not want to keep the mesh, we just want to get some information from it.
86  MCTexturedQuad * quad = [[MCMaterialController sharedMaterialController] quadFromAtlasKey:atlasKey];
87 
88  // get the material key so we can bind it during render
89  self.materialKey = quad.materialKey;
90 
91  // now we need to find the max/min of the UV coordinates
92  // this is the location in the atlas where our image is, and we are
93  // going to be applying it to every particle
94  CGFloat u,v;
95  NSInteger index;
96  minU = minV = 1.0;
97  maxU = maxV = 0.0;
98  CGFloat * uvs = [quad uvCoordinates];
99  for (index = 0; index < quad.vertexCount; index++) {
100  u = uvs[index * 2];
101  v = uvs[(index * 2) + 1];
102  if (u < minU) minU = u;
103  if (v < minV) minV = v;
104  if (u > maxU) maxU = u;
105  if (v > maxV) maxV = v;
106  }
107 }

143 -(void)buildVertexArrays
144 {
145  // go through all our individual particles and add their triangles
146  // to our big mesh
147  vertexIndex = 0;
148  for (MCParticle * particle in activeParticles) {
149  // 先刷新，更新粒子状态
150  [particle update]; // first, update each particle before we use it's data
151 
152  // check to see if we have run out of life, or are too small to see
153  // and if they are, then queue them for removal
154  if ((particle.life < 0) || (particle.size < 0.3)) {
155  [self removeChildParticle:particle];
156  continue; // skip to the next particle, no need to add this one
157  }
158 
159  // for each particle, need 2 triangles, so 6 verts
160  // first triangle of the quad. Need to load them in clockwise
161  // order since our models are in that order
162  [self addVertex:(particle.position.x - particle.size)
163  y:(particle.position.y + particle.size)
164  u:minU
165  v:minV];
166  [self addVertex:(particle.position.x + particle.size)
167  y:(particle.position.y - particle.size)
168  u:maxU
169  v:maxV];
170  [self addVertex:(particle.position.x - particle.size)
171  y:(particle.position.y - particle.size)
172  u:minU
173  v:maxV];
174 
175  // second triangle of the quad
176  [self addVertex:(particle.position.x - particle.size)
177  y:(particle.position.y + particle.size)
178  u:minU
179  v:minV];
180  [self addVertex:(particle.position.x + particle.size)
181  y:(particle.position.y + particle.size)
182  u:maxU
183  v:minV];
184  [self addVertex:(particle.position.x + particle.size)
185  y:(particle.position.y - particle.size)
186  u:maxU
187  v:maxV];
188 
189  }
190 }

246 -(void)emitNewParticles
247 {
248  if (!emit) return;
249  if (emitCounter > 0) emitCounter --; // if emitCounter == -1, then emit forever
250  if (emitCounter == 0) emit = NO; // this will be our last time through the emit method
251 
252  NSInteger newParticles = (NSInteger)MCRandomFloat(emissionRange); // grab a random number to be emitted
253 
254  NSInteger index;
255  CGFloat veloX,veloY;
256  // 一个个粒子准备发射
257  for (index = 0; index < newParticles; index++) {
258  if ([particlePool count] == 0) {
259  // if we are out of particles, then just get out early
260  return;
261  }
262  // grab a premade particle and set it up with some new random nubmers
263  MCParticle * p = [particlePool lastObject];
264  self.translation = MCPointMake(self.translation.x, self.translation.y, self.translation.z-0.001);
265  p.position = self.translation;
266  veloX = MCRandomFloat(xVelocityRange);
267  veloY = MCRandomFloat(yVelocityRange);
268  p.velocity = MCPointMake(veloX, veloY, -0.001);
269  p.life = MCRandomFloat(lifeRange);
270  p.size = MCRandomFloat(sizeRange);
271  p.grow = MCRandomFloat(growRange);
272  p.decay = MCRandomFloat(decayRange);
273 
274  // 同时更新粒子池和存活粒子数。
275  // add this particle
276  [activeParticles addObject:p];
277  // remove this particle from the unused array
278  [particlePool removeLastObject];
279  }

318 -(void)addVertex:(CGFloat)x y:(CGFloat)y u:(CGFloat)u v:(CGFloat)v
