138 - (void)gameLoop
139 {
140  // we use our own autorelease pool so that we can control when garbage gets collected
141  NSAutoreleasePool * apool = [[NSAutoreleasePool alloc] init];
142 
143  // 重置场景计时器
144  thisFrameStartTime = [levelStartDate timeIntervalSinceNow];
145  deltaTime = lastFrameStartTime - thisFrameStartTime;
146  lastFrameStartTime = thisFrameStartTime;
147 
148 
149 
150  // add any queued scene objects
151  if ([objectsToAdd count] > 0) {
152  [sceneObjects addObjectsFromArray:objectsToAdd];
153  [objectsToAdd removeAllObjects];
154  }
155 
156  // update our model
157  [self updateModel];
158 
159 
160  // deal with collisions
161  [collisionController handleCollisions];
162 
163  // send our objects to the renderer
164  [self renderScene];
165 
166  // remove any objects that need removal
167  if ([objectsToRemove count] > 0) {
168  [sceneObjects removeObjectsInArray:objectsToRemove];
169  [objectsToRemove removeAllObjects];
170  }
171 
172  [apool release];
173  if (needToLoadScene) [self loadScene];
174 }

225 -(void)setupLighting
226 {
227  // cull the unseen faces
228  // we use 'front' culling because exports our models to be compatible
229  // with this way
230  //glFrontFace(GL_CW);
231  // 仅仅处理我们能看到的前三个表面
232  glEnable(GL_CULL_FACE);
233  glCullFace(GL_BACK);
234 
235  // Light features
236  GLfloat light_ambient[]= { 2.0f, 2.0f, 2.0f, 1.0f };// 环境强度
237  GLfloat light_diffuse[]= { 80.0f, 80.0f, 80.0f, 1.0f }; // 散射光强度
238  GLfloat light_specular[]= { 10.0f, 10.0f, 10.0f, 1.0f };// 镜面光强度
239  // Set up light 0
240  glLightfv (GL_LIGHT0, GL_AMBIENT, light_ambient);
241  glLightfv (GL_LIGHT0, GL_DIFFUSE, light_diffuse);
242  glLightfv (GL_LIGHT0, GL_SPECULAR, light_specular);
243  glShadeModel (GL_SMOOTH);
244 
245  // Place the light up and to the right
246  // 如果该数组的最后一个值为1.0，表示指定的坐标为光源位置。如果是0.0，则表示光从无限远处沿指定的向量照射过来。
247  GLfloat light0_position[] = { 0.0, 0.0, 100.0, 1.0 };
248 
249  glLightfv(GL_LIGHT0, GL_POSITION, light0_position);
250

255  glEnable(GL_LIGHT0);
