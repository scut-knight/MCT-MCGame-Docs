158 - (BOOL)rotateOnAxis:(AxisType)axis onLayer:(int)layer inDirection:(LayerRotationDirectionType)direction{

160  MCCubie *layerCubes[9];

230  _orientationToMagicCubeFace[Front] = _orientationToMagicCubeFace[Right];
231  _orientationToMagicCubeFace[Right] = _orientationToMagicCubeFace[Back];
232  _orientationToMagicCubeFace[Back] = _orientationToMagicCubeFace[Left];
233  _orientationToMagicCubeFace[Left] = tmp;
234  } else {
235  FaceOrientationType tmp = _orientationToMagicCubeFace[Front];
236  _orientationToMagicCubeFace[Front] = _orientationToMagicCubeFace[Left];
237  _orientationToMagicCubeFace[Left] = _orientationToMagicCubeFace[Back];
238  _orientationToMagicCubeFace[Back] = _orientationToMagicCubeFace[Right];
239  _orientationToMagicCubeFace[Right] = tmp;
240  }

242  case Z:

245  _orientationToMagicCubeFace[Up] = _orientationToMagicCubeFace[Left];
246  _orientationToMagicCubeFace[Left] = _orientationToMagicCubeFace[Down];
247  _orientationToMagicCubeFace[Down] = _orientationToMagicCubeFace[Right];
248  _orientationToMagicCubeFace[Right] = tmp;
