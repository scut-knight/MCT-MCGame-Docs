190 - (MCUser *)queryUser:(NSString *)_name
191 {
192  [_name retain];
193 
194  MCUser *user = [MCUser alloc];
195  // 检查语句可以额外作为一个函数
196  if (sqlite3_open([[self dataFilePath] UTF8String], &database) != SQLITE_OK) {
197  sqlite3_close(database);
198  NSAssert(0,@"Failed to open");
199  }
200 
201  NSString *queryUserSQL = [[NSString alloc] initWithFormat:@"SELECT * FROM user WHERE name = '%@'",_name];
202 
203  sqlite3_stmt *stmt;// SQLite3的SQL语句类型
204  if (sqlite3_prepare_v2(database, [queryUserSQL UTF8String], -1, &stmt, nil) == SQLITE_OK) {
205  if (sqlite3_step(stmt) == SQLITE_ROW) {
206  //read a row
207  [user setUserID:sqlite3_column_int(stmt, 0)];// 只读第一个
208  // the _name may be changed by this
209  [user setName:[NSString stringWithUTF8String:((char*)sqlite3_column_text(stmt, 1))]];
210  [user setSex:[NSString stringWithUTF8String:((char*)sqlite3_column_text(stmt, 2))]];
211  [user setTotalMoves:sqlite3_column_int(stmt, 3)];
212  [user setTotalGameTime:sqlite3_column_double(stmt, 4)];
213  [user setTotalLearnTime:sqlite3_column_double(stmt, 5)];
214  [user setTotalFinish:sqlite3_column_int(stmt, 6)];
215  }
216  else {
217  [user initWithUserID:0 UserName:nil UserSex:nil totalMoves:0 totalGameTime:0 totalLearnTime:0 totalFinish:0];
218  NSAssert(0,@"Failed to fetch row");
219  }
220 
221  }
222  else {
223  [user initWithUserID:0 UserName:nil UserSex:nil totalMoves:0 totalGameTime:0 totalLearnTime:0 totalFinish:0];
224  NSAssert(0,@"Failed to select");
225  }
226 
227  sqlite3_finalize(stmt);
228  sqlite3_close(database);
229  [queryUserSQL release];
230  [_name release];
231 
232  NSLog(@"query user success");
233 
234  return [user autorelease];
