104 int CubieCube::Cnk(int n, int k){
105  int i, j, s;
106  if (n < k)
107  return 0;
108  if (k > n / 2)
109  k = n - k;
110  for (s = 1, i = n, j = 1; i != n - k; i--, j++) {
111  s *= i;
112  s /= j;
113  }
114  return s;
115 }

192 void CubieCube::cornerMultiply(CubieCube &b) {
193  Corner cPerm[8] = {URF};
194  unsigned char cOri[8] = {0};
195  for (int i = 0; i < 8; i++) {
196  cPerm[i] = cp[b.cp[i]];
197  unsigned char oriA = co[b.cp[i]];
198  unsigned char oriB = b.co[i];
199 
200  unsigned char ori = 0;
201 
202  if (oriA < 3 && oriB < 3) // if both cubes are regular cubes...
203  {
204  ori = (unsigned char) (oriA + oriB); // just do an addition modulo 3 here
205  if (ori >= 3)
206  ori -= 3; // the composition is a regular cube
207 
208  // +++++++++++++++++++++not used in this implementation +++++++++++++++++++++++++++++++++++
209  } else if (oriA < 3 && oriB >= 3) // if cube b is in a mirrored
210  // state...
211  {
212  ori = (unsigned char) (oriA + oriB);
213  if (ori >= 6)
214  ori -= 3; // the composition is a mirrored cube
215  } else if (oriA >= 3 && oriB < 3) // if cube a is an a mirrored
216  // state...
217  {
218  ori = (unsigned char) (oriA - oriB);
219  if (ori < 3)
220  ori += 3; // the composition is a mirrored cube
221  } else if (oriA >= 3 && oriB >= 3) // if both cubes are in mirrored
222  // states...
223  {
224  ori = (unsigned char) (oriA - oriB);
225  if (ori < 0)
226  ori += 3; // the composition is a regular cube
227  // ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
228  }
229  cOri[i] = ori;
230  }
231 
232  for (int i = 0; i < 8; i++) {
233  cp[i] = cPerm[i];
234  co[i] = cOri[i];
235  }

366  int a = 0, x = 0;
367  Edge edge4[4];
368  // compute the index a < (12 choose 4) and the permutation array perm.
369  for (int j = BR; j >= UR; j--)
370  if (FR <= ep[j] && ep[j] <= BR) {
371  a += Cnk(11 - j, x + 1);
372  edge4[3 - x++] = ep[j];

381  rotateLeft(edge4, 0, j);

391  int x;
392  Edge sliceEdge[] = { FR, FL, BL, BR };
393  Edge otherEdge[] = { UR, UF, UL, UB, DR, DF, DL, DB };
394  int b = idx % 24; // Permutation
395  int a = idx / 24; // Combination
396 
397  for (int i = 0; i < 12; i++) {
398  ep[i] = DB;// Use UR to invalidate all edges
399  }
400 
401  for (int j = 1, k; j < 4; j++)// generate permutation from index b
402  {
403  k = b % (j + 1);
404  b /= j + 1;
405  while (k-- > 0)
406  rotateRight(sliceEdge, 0, j);
407  }
408 
409  x = 3;// generate combination and set slice edges
410  for (int j = UR; j <= BR; j++)
411  if (a - Cnk(11 - j, x + 1) >= 0) {
412  ep[j] = sliceEdge[3 - x];
413  a -= Cnk(11 - j, x-- + 1);
414  }
415  x = 0; // set the remaining edges UR..DB
416  for (int j = UR; j <= BR; j++)
417  if (ep[j] == DB)
418  ep[j] = otherEdge[x++];
