18 + (NSMutableArray *)avaiableColorsOfCubie:(NSObject<MCCubieDelegate> *)cubie
19  inOrientation:(FaceOrientationType)orientation{
20  // If cubie has no colors, all colors are avaiable.
21  BOOL counterTable[6] = {YES, YES, YES, YES, YES, YES};
22 
23  // All avaiable colors will be here.
24  NSMutableArray *avaiableColors = [NSMutableArray arrayWithCapacity:6];
25 
26  // According to colors owned by cubie, remove some avaiable colors.
27  NSArray *allColors = [cubie allFaceColors];
28  NSArray *allOrientations = [cubie allOrientations];
29 
30  // num of completed colors
31  int completedColorsNum = 0;
32  int skinNum = [cubie skinNum];
33  for (int i = 0; i < skinNum; i++) {
34  // If the orientation is target orientation, ignore it.
35  FaceOrientationType faceOrientation = (FaceOrientationType)[[allOrientations objectAtIndex:i] integerValue];
36  if (faceOrientation == orientation) continue;
37 
38  // If it's NoColor, ignore it.
39  FaceColorType faceColor = (FaceColorType)[[allColors objectAtIndex:i] integerValue];
40  if (faceColor == NoColor) continue;
41 
42  // Increase num of completed colors
43  completedColorsNum++;
44  // 翻转所有面
45  for (int j = 0; j < 6; j++) {
46  // flip all the valuable colors
47  counterTable[j] = !counterTable[j];
48  }
49  // 当前面(已经上色了的)和看不见的邻接面都会被翻转一次，准确来说，会被复原
50  if (faceColor % 2 == 0) { // Up Front Left
51  counterTable[faceColor+1] = !counterTable[faceColor+1];
52  }
53  else{ // Down Back Right
54  counterTable[faceColor-1] = !counterTable[faceColor-1];
55  }
56  counterTable[faceColor] = !counterTable[faceColor];
57  }
58  // 最多有三个面已经填充了颜色，而不考虑当前面，所以completedColorsNum为0-2
59  switch (completedColorsNum) {
60  case 0: // all are available
61  {
62  for (int i = 0; i < 6; i++) {
63  [avaiableColors addObject:[NSNumber numberWithInteger:i]];
64  }
65  }
66  break;
67  case 1:
68  {
69  for (int i = 0; i < 6; i++) {
70  // 看得见的而且没有上色的面
71  if (counterTable[i] == NO) {
72  [avaiableColors addObject:[NSNumber numberWithInteger:i]];
73  }
74  }
75  }
76  break;
77  case 2:
78  {
79  for (int i = 0; i < 6; i++) {
80  // 翻转了两次后看得见的而且没有上色的面
81  if (counterTable[i] == YES) {
82  [avaiableColors addObject:[NSNumber numberWithInteger:i]];
83  }
84  }
85  }
86  break;
87  default:
88  break;
89  }
90 
91 
92  return avaiableColors;
93 }
