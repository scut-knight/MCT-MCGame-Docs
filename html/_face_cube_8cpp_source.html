131 std::auto_ptr<CubieCube> FaceCube::toCubieCube(){
132  std::auto_ptr<CubieCube> ccRet(new CubieCube());
133 
134  unsigned char ori;
135  for (int i = 0; i < 8; i++)
136  ccRet->cp[i] = URF;// invalidate corners
137  for (int i = 0; i < 12; i++)
138  ccRet->ep[i] = UR;// and edges
139  Color col1, col2;
140 
141  for (int i = 0; i < 8; i++) {
142  // get the colors of the cubie at corner i, starting with U/D
143  for (ori = 0; ori < 3; ori++)
144  if (f[cornerFacelet[i][ori]] == U || f[cornerFacelet[i][ori]] == D)
145  break;
146  col1 = f[cornerFacelet[i][(ori + 1) % 3]];
147  col2 = f[cornerFacelet[i][(ori + 2) % 3]];
148 
149  for (int j = 0; j < 8; j++) {
150  if (col1 == cornerColor[j][1] && col2 == cornerColor[j][2]) {
151  // in cornerposition i we have cornercubie j
152  ccRet->cp[i] = (Corner)j;
153  ccRet->co[i] = (unsigned char) (ori % 3);
154  break;
155  }
156  }
157 
158  }
159 
160  for (int i = 0; i < 12; i++) {
161  for (int j = 0; j < 12; j++) {
162  if (f[edgeFacelet[i][0]] == edgeColor[j][0]
163  && f[edgeFacelet[i][1]] == edgeColor[j][1]) {
164  ccRet->ep[i] = (Edge)j;
165  ccRet->eo[i] = 0;
166  break;
167  }
168  if (f[edgeFacelet[i][0]] == edgeColor[j][1]
169  && f[edgeFacelet[i][1]] == edgeColor[j][0]) {
170  ccRet->ep[i] = (Edge)j;
171  ccRet->eo[i] = 1;
172  break;
173  }
174  }
175 
176  }
177 
178 
179  return ccRet;
180 
181 }
