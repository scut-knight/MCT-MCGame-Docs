147 void CoordCube::move(int m) {
148  twist = twistMove[twist][m];
149  flip = flipMove[flip][m];
150  parity = parityMove[parity][m];
151  FRtoBR = FRtoBR_Move[FRtoBR][m];
152  URFtoDLF = URFtoDLF_Move[URFtoDLF][m];
153  URtoUL = URtoUL_Move[URtoUL][m];
154  UBtoDF = UBtoDF_Move[UBtoDF][m];
155  if (URtoUL < 336 && UBtoDF < 336)// updated only if UR,UF,UL,UB,DR,DF
156  // are not in UD-slice
157  URtoDF = MergeURtoULandUBtoDF[URtoUL][UBtoDF];
158 }
