115 string Search::solution(string facelets, int maxDepth, long timeOut, bool useSeparator) {
116  int s = 0;
117 
118  // +++++++++++++++++++++check for wrong input +++++++++++++++++++++++++++++
119  int count[6] = {0};
120  try {
121  for (int i = 0; i < 54; i++)
122  {
123  switch (facelets[i]) {
124  case 'U':
125  count[U]++;
126  break;
127  case 'R':
128  count[R]++;
129  break;
130  case 'F':
131  count[F]++;
132  break;
133  case 'D':
134  count[D]++;
135  break;
136  case 'L':
137  count[L]++;
138  break;
139  case 'B':
140  count[B]++;
141  break;
142  default:
143  break;
144  }
145  }
146  } catch (exception e) {
147  return "Error 1"; // here is not exactly one facelet of each colour
148  }
149  for (int i = 0; i < 6; i++)
150  if (count[i] != 9)
151  return "Error 1";
152 
153  FaceCube fc = FaceCube(facelets);
154  auto_ptr<CubieCube> cc = fc.toCubieCube();
155  if ((s = cc->verify()) != 0){
156  string result = "Error ";
157  result += (char)(abs(s)+'0');
158  return result;
159  }
160 
161  // +++++++++++++++++++++++ initialization +++++++++++++++++++++++++++++++++
162  //---------------------------
163  CoordCube c = CoordCube(*cc);
164 
165  po[0] = 0;
166  ax[0] = 0;
167  flip[0] = c.flip;
168  twist[0] = c.twist;
169  slice[0] = c.FRtoBR / 24;
170  URFtoDLF[0] = c.URFtoDLF;
171  FRtoBR[0] = c.FRtoBR;
172  URtoUL[0] = c.URtoUL;
173  UBtoDF[0] = c.UBtoDF;
174 
175  minDistPhase1[1] = 1;// else failure for depth=1, n=0
176  int mv = 0, n = 0;
177  bool busy = false;
178  int depthPhase1 = 1;
179 
180  struct timeval tv;
181  gettimeofday(&tv,NULL);
182 
183  long tStart = tv.tv_sec * 1000 + tv.tv_usec / 1000;
184 
185  // +++++++++++++++++++ Main loop ++++++++++++++++++++++++++++++++++++++++++
186  do {/*0*/
187  do {/*1*/
188  if ((depthPhase1 - n > minDistPhase1[n + 1]) && !busy) {
189 
190  if (ax[n] == 0 || ax[n] == 3)// Initialize next move
191  ax[++n] = 1;
192  else
193  ax[++n] = 0;
194  po[n] = 1;
195  } else if (++po[n] > 3) {
196  /*2*/do {// increment axis
197  if (++ax[n] > 5) {
198 
199  gettimeofday(&tv,NULL);
200 
201  if ((tv.tv_sec * 1000 + tv.tv_usec / 1000) - tStart > timeOut << 10)
202  return "Error 8"; //Timeout, no solution within given time
203 
204  if (n == 0) {
205  if (depthPhase1 >= maxDepth)
206  return "Error 7";// No solution exists for the given maxDepth
207  else {
208  depthPhase1++;
209  ax[n] = 0;
210  po[n] = 1;
211  busy = false;
212  break;
213  }
214  } else {
215  n--;
216  busy = true;
217  break;
218  }
219 
220  } else {
221  po[n] = 1;
222  busy = false;
223  }
224  /*2*/ } while (n != 0 && (ax[n - 1] == ax[n] || ax[n - 1] - 3 == ax[n]));
225  } else
226  busy = false;
227  } while (busy);/*1*/
228 
229  // +++++++++++++ compute new coordinates and new minDistPhase1 ++++++++++
230  // if minDistPhase1 =0, the H subgroup is reached
231  mv = 3 * ax[n] + po[n] - 1;
232  flip[n + 1] = CoordCube::flipMove[flip[n]][mv];
233  twist[n + 1] = CoordCube::twistMove[twist[n]][mv];
234  slice[n + 1] = CoordCube::FRtoBR_Move[slice[n] * 24][mv] / 24;
235  minDistPhase1[n + 1] = max(CoordCube::getPruning(CoordCube::Slice_Flip_Prun, CoordCube::N_SLICE1 * flip[n + 1]
236  + slice[n + 1]),
237  CoordCube::getPruning(CoordCube::Slice_Twist_Prun, CoordCube::N_SLICE1 * twist[n + 1]
238  + slice[n + 1]));
239  // ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
240 
241  if (minDistPhase1[n + 1] == 0 && n >= depthPhase1 - 5) {
242  minDistPhase1[n + 1] = 10;// instead of 10 any value >5 is possible
243  if (n == depthPhase1 - 1 && (s = totalDepth(depthPhase1, maxDepth)) >= 0) {
244  if (s == depthPhase1
245  || (ax[depthPhase1 - 1] != ax[depthPhase1] && ax[depthPhase1 - 1] != ax[depthPhase1] + 3))
246  return useSeparator ? solutionToString(s, depthPhase1) : solutionToString(s);
247  }
248 
249  }
250  } while (true);/*0*/
251 }

254 int Search::totalDepth(int depthPhase1, int maxDepth) {
255  int mv = 0, d1 = 0, d2 = 0;
256  int maxDepthPhase2 = min(10, maxDepth - depthPhase1);// Allow only max 10 moves in phase2
257  for (int i = 0; i < depthPhase1; i++) {
258  mv = 3 * ax[i] + po[i] - 1;
259  URFtoDLF[i + 1] = CoordCube::URFtoDLF_Move[URFtoDLF[i]][mv];
260  FRtoBR[i + 1] = CoordCube::FRtoBR_Move[FRtoBR[i]][mv];
261  parity[i + 1] = CoordCube::parityMove[parity[i]][mv];
262  }
263 
264  if ((d1 = CoordCube::getPruning(CoordCube::Slice_URFtoDLF_Parity_Prun,
265  (CoordCube::N_SLICE2 * URFtoDLF[depthPhase1]
266  + FRtoBR[depthPhase1]) * 2
267  + parity[depthPhase1])
268  ) > maxDepthPhase2)
269  return -1;
270 
271  for (int i = 0; i < depthPhase1; i++) {
272  mv = 3 * ax[i] + po[i] - 1;
273  URtoUL[i + 1] = CoordCube::URtoUL_Move[URtoUL[i]][mv];
274  UBtoDF[i + 1] = CoordCube::UBtoDF_Move[UBtoDF[i]][mv];
275  }
276  URtoDF[depthPhase1] = CoordCube::MergeURtoULandUBtoDF[URtoUL[depthPhase1]][UBtoDF[depthPhase1]];
277 
278  if ((d2 = CoordCube::getPruning(CoordCube::Slice_URtoDF_Parity_Prun,
279  (CoordCube::N_SLICE2 * URtoDF[depthPhase1]
280  + FRtoBR[depthPhase1]) * 2
281  + parity[depthPhase1])
282  ) > maxDepthPhase2)
283  return -1;
284  // if max(d1,d2) == 0
285  if ((minDistPhase2[depthPhase1] = max(d1, d2)) == 0)// already solved
286  return depthPhase1;
287 
288  // now set up search
289 
290  int depthPhase2 = 1;
291  int n = depthPhase1;
292  bool busy = false;
293  po[depthPhase1] = 0;
294  ax[depthPhase1] = 0;
295  minDistPhase2[n + 1] = 1;// else failure for depthPhase2=1, n=0
296  // +++++++++++++++++++ end initialization +++++++++++++++++++++++++++++++++
297  do {
298  do {
299  if ((depthPhase1 + depthPhase2 - n > minDistPhase2[n + 1]) && !busy) {
300 
301  if (ax[n] == 0 || ax[n] == 3)// Initialize next move
302  {
303  ax[++n] = 1;
304  po[n] = 2;
305  } else {
306  ax[++n] = 0;
307  po[n] = 1;
308  }
309  } else if ((ax[n] == 0 || ax[n] == 3) ? (++po[n] > 3) : ((po[n] = po[n] + 2) > 3)) {
310  do {// increment axis
311  if (++ax[n] > 5) {
312  if (n == depthPhase1) {
313  if (depthPhase2 >= maxDepthPhase2)
314  return -1;
315  else {
316  depthPhase2++;
317  ax[n] = 0;
318  po[n] = 1;
319  busy = false;
320  break;
321  }
322  } else {
323  n--;
324  busy = true;
325  break;
326  }
327 
328  } else {
329  if (ax[n] == 0 || ax[n] == 3)
330  po[n] = 1;
331  else
332  po[n] = 2;
333  busy = false;
334  }
335  } while (n != depthPhase1 && (ax[n - 1] == ax[n] || ax[n - 1] - 3 == ax[n]));
336  } else
337  busy = false;
338  } while (busy);
339  // +++++++++++++ compute new coordinates and new minDist ++++++++++
340  mv = 3 * ax[n] + po[n] - 1;
341 
342  URFtoDLF[n + 1] = CoordCube::URFtoDLF_Move[URFtoDLF[n]][mv];
343  FRtoBR[n + 1] = CoordCube::FRtoBR_Move[FRtoBR[n]][mv];
344  parity[n + 1] = CoordCube::parityMove[parity[n]][mv];
345  URtoDF[n + 1] = CoordCube::URtoDF_Move[URtoDF[n]][mv];
346 
347  minDistPhase2[n + 1] = max(CoordCube::getPruning(CoordCube::Slice_URtoDF_Parity_Prun,
348  (CoordCube::N_SLICE2 * URtoDF[n + 1] + FRtoBR[n + 1])
349  * 2 + parity[n + 1]),
350  CoordCube::getPruning(CoordCube::Slice_URFtoDLF_Parity_Prun,
351  (CoordCube::N_SLICE2 * URFtoDLF[n + 1] + FRtoBR[n + 1])
352  * 2 + parity[n + 1]));
353  // ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
354 
355  } while (minDistPhase2[n + 1] != 0);
356 
357  return depthPhase1 + depthPhase2;
358 }
