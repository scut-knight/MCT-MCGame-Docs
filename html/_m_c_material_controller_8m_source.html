297 -(CGSize)loadTextureImage:(NSString*)imageName materialKey:(NSString*)materialKey
298 {
299  CGContextRef spriteContext;
300  GLubyte *spriteData;
301  size_t width, height;
302 
303 
304  GLuint textureID;
305  // grab the image off the file system, jam it into a CGImageRef
306 
307  UIImage* uiImage = [[UIImage alloc] initWithContentsOfFile:[[NSBundle mainBundle] pathForResource:imageName ofType:nil]];
308  CGImageRef spriteImage = [uiImage CGImage];
309  // Get the width and height of the image
310  width = CGImageGetWidth(spriteImage);
311  height = CGImageGetHeight(spriteImage);
312 
313  CGSize imageSize = CGSizeMake(width, height);
314  // Texture dimensions must be a power of 2. If you write an application that allows users to supply an image,
315  // you'll want to add code that checks the dimensions and takes appropriate action if they are not a power of 2.
316 
317  if(spriteImage) {
318  // Allocated memory needed for the bitmap context
319  spriteData = (GLubyte *) malloc(width * height * 4);
320  memset(spriteData, 0, (width * height * 4));
321  // Uses the bitmatp creation function provided by the Core Graphics framework.
322  spriteContext = CGBitmapContextCreate(spriteData, width, height, 8, width * 4, CGImageGetColorSpace(spriteImage), kCGImageAlphaPremultipliedLast);
323  // After you create the context, you can draw the sprite image to the context.
324  CGContextDrawImage(spriteContext, CGRectMake(0.0, 0.0, (CGFloat)width, (CGFloat)height), spriteImage);
325  // You don't need the context at this point, so you need to release it to avoid memory leaks.
326  CGContextRelease(spriteContext);
327 
328  // Use OpenGL ES to generate a name for the texture.
329  glGenTextures(1, &textureID);
330  // Bind the texture name.
331  glBindTexture(GL_TEXTURE_2D, textureID);
332  // Specidfy a 2D texture image, provideing the a pointer to the image data in memory
333 
334  //Convert "RRRRRRRRRGGGGGGGGBBBBBBBBAAAAAAAA" to "RRRRRGGGGBBBBAAAA"
335  if (MC_CONVERT_TO_4444) {
336  void* tempData;
337  unsigned int* inPixel32;
338  unsigned short* outPixel16;
339 
340  tempData = malloc(height * width * 2);
341 
342  inPixel32 = (unsigned int*)spriteData;
343  outPixel16 = (unsigned short*)tempData;
344  NSUInteger i;
345  for(i = 0; i < width * height; ++i, ++inPixel32)
346  *outPixel16++ =
347  ((((*inPixel32 >> 0) & 0xFF) >> 4) << 12) |
348  ((((*inPixel32 >> 8) & 0xFF) >> 4) << 8) |
349  ((((*inPixel32 >> 16) & 0xFF) >> 4) << 4) |
350  ((((*inPixel32 >> 24) & 0xFF) >> 4) << 0);
351 
352  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_SHORT_4_4_4_4, tempData);
353  free(tempData);
354  } else {
355 
356  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, spriteData);
357  }
358 
359  free(spriteData);
360  // Release the image data
361  // Set the texture parameters to use a minifying filter and a linear filer (weighted average)
362  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
363  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
364 
365  // Enable use of the texture
366  glEnable(GL_TEXTURE_2D);
367  // Set a blending function to use
368  glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
369  // Enable blending
370  glEnable(GL_BLEND);
371  } else {
372  [uiImage release];
373  return CGSizeZero;
374  }
375  [uiImage release];
376 
377  if (materialLibrary == nil) materialLibrary = [[NSMutableDictionary alloc] init];
378 
379  // now put the texture ID into the library
380  [materialLibrary setObject:[NSNumber numberWithUnsignedInt:textureID] forKey:materialKey];
381  return imageSize;
382 }
