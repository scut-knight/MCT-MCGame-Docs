44 -(MCPoint)calculateCentroid
45 {
46  CGFloat xTotal = 0;
47  CGFloat yTotal = 0;
48  CGFloat zTotal = 0;
49  NSInteger index;
50  // step through each vertex and add them all up
51  for (index = 0; index < vertexCount; index++) {
52  NSInteger position = index * vertexSize;
53  xTotal += vertexes[position];
54  yTotal += vertexes[position + 1];
55  if (vertexSize > 2) zTotal += vertexes[position + 2];
56  }
57  // now average each total over the number of vertexes
58  // 注意z轴坐标上的值恒为0.0
59  return MCPointMake(xTotal/(CGFloat)vertexCount, yTotal/(CGFloat)vertexCount, 0.0);
60 }

67 -(CGFloat)calculateRadius

125  }
126  CGRect meshBounds = CGRectMake(xMin, yMin, xMax - xMin, yMax - yMin);
