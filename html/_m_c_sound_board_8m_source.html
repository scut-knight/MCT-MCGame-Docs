140 - (void)fadeIn:(NSTimer *)timer
141 {
142  backgroundPlayer = [timer.userInfo objectForKey:@"player"];
143  float maxvolume = [[timer.userInfo objectForKey:@"maxvolume"]floatValue];;
144  float volume = backgroundPlayer.volume;
145  volume = volume + 1.0 / MCSOUNDBOARD_AUDIO_FADE_STEPS;
146  volume = volume > maxvolume ? maxvolume : volume;

161 - (void)playAudioForKey:(id)key fadeInInterval:(NSTimeInterval)fadeInInterval maxVolume:(NSNumber*)maxvolume
162 {

164  if (fadeInInterval > 0.0) {
165  backgroundPlayer.volume = 0.0;
166  NSMutableDictionary *dic = [[NSMutableDictionary alloc]init];
167  [dic setObject:backgroundPlayer forKey:@"player"];
168  [dic setObject:maxvolume forKey:@"maxvolume"];

170  [NSTimer scheduledTimerWithTimeInterval:interval
171  target:self
172  selector:@selector(fadeIn:)
173  userInfo:dic
174  repeats:YES];
175  [dic release];
176  }
