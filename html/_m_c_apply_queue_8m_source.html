53  for (MCTreeNode *child in action.children) {
54  if (child.value == Fw2 || child.value == Bw2 || child.value == Rw2 || child.value == Lw2 ||
55  child.value == Uw2 || child.value == Dw2 ) {
56  // 注意以上的旋转动作是由两个动作合起来的(它们的后缀都有2)
57  // 所以拆分成两个再加入队列中
58  [self.rotationQueue addObject:[NSNumber numberWithInteger:(child.value-2)]];
59  [self.rotationQueue addObject:[NSNumber numberWithInteger:(child.value-2)]];
60  }
61  else{
62  [self.rotationQueue addObject:[NSNumber numberWithInteger:child.value]];
63  }
64  }
65  }
66  break;
67  case FaceToOrientation:

74  elementNode = [action.children objectAtIndex:1];

104 - (void)applyRotation:(SingmasterNotation)currentRotation{
105  //detect the rotation result
106  //if the queue is exist, continue
107  //else, return finished
108  if (self.rotationQueue == nil) {
109  previousResult = Finished;
110  }
111  else{
112  //if they are same, accord
113  SingmasterNotation targetRotation = (SingmasterNotation)[[self.rotationQueue objectAtIndex:currentRotationQueuePosition] integerValue];
114 
115  if (previousResult == StayForATime) {
116  // 如果不需要执行旋转
117  if ([MCCompositeRotationUtil isSingmasterNotation:previousRotation andSingmasterNotation:currentRotation equalTo:targetRotation]) {
118  previousResult = Accord;
119  currentRotationQueuePosition++;
120  }
121  else{
122  previousResult = Disaccord;
123  //extra queue
124  [self.extraRotations removeAllObjects];
125  //注意在extraRotations中，previousRotation在currentRotation之后
126  [self.extraRotations addObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:currentRotation]]];
127  [self.extraRotations addObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:previousRotation]]];
128  //self queue
129  [self.rotationQueue insertObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:currentRotation]] atIndex:currentRotationQueuePosition];
130  [self.rotationQueue insertObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:previousRotation]] atIndex:currentRotationQueuePosition];
131  }
132 
133  } else { // else 0
134  if (targetRotation == currentRotation) {
135  previousResult = Accord;
136  currentRotationQueuePosition++;
137  if ([self isFinished]) previousResult = Finished;
138  }
139  else{ // else 1
140  if ([MCCompositeRotationUtil isSingmasterNotation:currentRotation PossiblePartOfSingmasterNotation:targetRotation]) {
141  previousResult = StayForATime;
142  }
143  else{ //else 2
144  previousResult = Disaccord;
145  //extra queue
146  [self.extraRotations removeAllObjects];
147  [self.extraRotations addObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:currentRotation]]];
148  //self queue
149  [self.rotationQueue insertObject:[NSNumber numberWithInteger:[MCTransformUtil getContrarySingmasterNotation:currentRotation]] atIndex:currentRotationQueuePosition];
150  }// else 2
151  }// else 1
152  }// else 0
153 
154  if (currentRotationQueuePosition == [self.rotationQueue count]) {
155  self.rotationQueue = nil;
156  previousResult = Finished;
157  // 调用onQueueComplete结束对queue的处理
158  [self.queueCompleteDelegate onQueueComplete];
159  }
160  }
161  previousRotation = currentRotation;
162 }
