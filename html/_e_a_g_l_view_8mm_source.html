213 -(void)perspectiveFovY:(GLfloat)fovY
214  aspect:(GLfloat)aspect
215  zNear:(GLfloat)zNear
216  zFar:(GLfloat)zFar
217 {
218  const GLfloat pi = 3.1415926;
219  // Half of the size of the x and y clipping planes.
220  // - halfWidth = left, halfWidth = right
221  // - halfHeight = bottom, halfHeight = top
222  GLfloat halfWidth, halfHeight;
223  // Calculate the distance from 0 of the y clipping plane. Basically trig to calculate
224  // position of clip plane at zNear.
225  halfHeight = tan( (fovY / 2) / 180 * pi ) * zNear;
226  // Calculate the distance from 0 of the x clipping plane based on the aspect ratio.
227  halfWidth = halfHeight * aspect;
228  // Finally call glFrustum with our calculated values.
229  glFrustumf( -halfWidth, halfWidth, -halfHeight, halfHeight, zNear, zFar );
230 }
