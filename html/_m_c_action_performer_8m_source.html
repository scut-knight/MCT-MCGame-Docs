118 - (NSInteger)treeNodesApply:(MCTreeNode *)root{
119  NSObject<MCMagicCubeDelegate> *magicCube = self.workingMemory.magicCube;
120 
121  switch (root.type) {
122  case ExpNode:
123  {
124  if (root.value == Sequence) {
125  return [self sequenceNodeApply:root];
126  }
127  }
128  break;
129  // 动作
130  case ActionNode:
131  {
132  switch (root.value) {
133  case Rotate:
134  for (MCTreeNode *child in root.children) {
135  [magicCube rotateWithSingmasterNotation:(SingmasterNotation)child.value];
136  }
137  break;
138  case FaceToOrientation:
139  {
140  MCTreeNode *elementNode;
141  elementNode = [root.children objectAtIndex:0];
142  ColorCombinationType targetCombination = (ColorCombinationType)elementNode.value;
143  struct Point3i targetCoor = [magicCube coordinateValueOfCubieWithColorCombination:targetCombination];
144  elementNode = [root.children objectAtIndex:1];
145  FaceOrientationType targetOrientation = (FaceOrientationType)elementNode.value;
146  [magicCube rotateWithSingmasterNotation:[MCTransformUtil getPathToMakeCenterCubieAtPosition:targetCoor inOrientation:targetOrientation]];
147  }
148  break;
149  case LockCubie:
150  {
151  MCTreeNode *elementNode = [root.children objectAtIndex:0];
152  int index = 0;
153  if ([root.children count] != 1) {
154  index = [(MCTreeNode *)[root.children objectAtIndex:1] value];
155  }
156  int identity = [self treeNodesApply:elementNode];
157  if (identity == -1) {
158  [self.workingMemory unlockCubieAtIndex:index];
159  }
160  else{
161  [self.workingMemory lockCubie:[magicCube cubieWithColorCombination:(ColorCombinationType)identity]
162  atIndex:index];
163  }
164  }
165  break;
166  case UnlockCubie:
167  {
168  int index = 0;
169  if ([root.children count] != 0) {
170  index = [(MCTreeNode *)[root.children objectAtIndex:0] value];
171  }
172  [self.workingMemory unlockCubieAtIndex:index];
173  }
174  break;
175  default:
176  return NO;
177  }
178  }
179  break;
180  // 信息
181  case InformationNode:
182  switch (root.value) {
183  case getCombinationFromOrientation:
184  {
185  int x=1, y=1, z=1;
186  for (MCTreeNode *child in root.children) {
187  switch ([magicCube centerMagicCubeFaceInOrientation:(FaceOrientationType)child.value]) {
188  case Up:
189  y = 2;
190  break;
191  case Down:
192  y = 0;
193  break;
194  case Left:
195  x = 0;
196  break;
197  case Right:
198  x = 2;
199  break;
200  case Front:
201  z = 2;
202  break;
203  case Back:
204  z = 0;
205  break;
206  default:
207  break;
208  }
209  }
210 
211  //Store the result at the node
212  root.result = x+y*3+z*9;
213  return root.result;
214  }
215  case getCombinationFromColor:
216  {
217  // 跟getCombinationFromOrientation相同的操作
218  int x=1, y=1, z=1;
219  for (MCTreeNode *child in root.children) {
220  switch ((FaceColorType)[self treeNodesApply:child]) {
221  case UpColor:
222  y = 2;
223  break;
224  case DownColor:
225  y = 0;
226  break;
227  case LeftColor:
228  x = 0;
229  break;
230  case RightColor:
231  x = 2;
232  break;
233  case FrontColor:
234  z = 2;
235  break;
236  case BackColor:
237  z = 0;
238  break;
239  default:
240  break;
241  }
242  }
243 
244  //Store the result at the node
245  root.result = x+y*3+z*9;
246  return root.result;
247  }
248  case getFaceColorFromOrientation:
249  {
250  FaceColorType color;
251  FaceOrientationType orientation = (FaceOrientationType)[(MCTreeNode *)[root.children objectAtIndex:0] value];
252  if ([root.children count] == 1) {
253  switch ([magicCube centerMagicCubeFaceInOrientation:orientation]) {
254  case Up:
255  color = UpColor;
256  break;
257  case Down:
258  color = DownColor;
259  break;
260  case Left:
261  color = LeftColor;
262  break;
263  case Right:
264  color = RightColor;
265  break;
266  case Front:
267  color = FrontColor;
268  break;
269  case Back:
270  color = BackColor;
271  break;
272  default:
273  color = NoColor;
274  break;
275  }
276  }
277  else{
278  // 如果count != 1，属性储存在第二个子节点中
279  int value = [(MCTreeNode *)[root.children objectAtIndex:1] value];
280  struct Point3i coordinate = {value%3-1, value%9/3-1, value/9-1};
281  color = [[magicCube cubieAtCoordinatePoint3i:coordinate] faceColorInOrientation:orientation];
282  }
283 
284  //Store the result at the node
285  root.result = color;
286  return root.result;
287  }
288  case lockedCubie:
289  {
290  int index = 0;
291  if ([root.children count] != 0) {
292  index = [(MCTreeNode *)[root.children objectAtIndex:0] value];
293  }
294 
295  if ([self.workingMemory lockerEmptyAtIndex:index]) {
296  root.result = -1;
297  }
298  else{
299  root.result = [[self.workingMemory cubieLockedInLockerAtIndex:index] identity];
300  }
301 
302  return root.result;
303  }
304  default:
305  break;
306  }
307  break;
308  case ElementNode:
309  return root.value;
310  default:
311  return NO;
312  }
313  return YES;
314 }

331 - (BOOL)decomposeRule:(MCRule *)rule{
332  // Get the tree of the action according to the pattern name
333  MCTreeNode *actionTree = [rule root];
334 
335  // Analyse the action and return the result
336  switch (actionTree.type) {
337  // 运算符节点的情况
338  case ExpNode:
339  {
340  BOOL flag = YES;
341  for (MCTreeNode *node in actionTree.children) {
342  if (flag) {
343  if (node.type == ActionNode && (node.value == Rotate|| node.value == FaceToOrientation)) {
344  // 应用队列来进行处理。这里终于要初始化MCWorkingMemory的applyQueue成员
345  self.workingMemory.applyQueue = [MCApplyQueue applyQueueWithRotationAction:node withMagicCube:self.workingMemory.magicCube];
346  self.workingMemory.applyQueue.queueCompleteDelegate = self;
347  flag = NO;
348  }
349  else{// not special type
350  [self treeNodesApply:node];
351  }
352  } else {// flag == NO
353  //符合ActionNode && (Rotate || FaceToOrientation)的节点之后的剩余节点
354  [self.workingMemory.residualActions addObject:node];
355  }
356  }
357  }
358  break;
359  // 动作节点的情况。可以直接处理了。
360  case ActionNode:
361  switch (actionTree.value) {
362  case Rotate:
363  // fall through
364  case FaceToOrientation:
365  self.workingMemory.applyQueue = [MCApplyQueue applyQueueWithRotationAction:actionTree withMagicCube:self.workingMemory.magicCube];
366  self.workingMemory.applyQueue.queueCompleteDelegate = self;
367  break;
368  default:
369  [self treeNodesApply:actionTree];
370  break;
371  }
372  break;
373 
374  default:
375  return NO;
376  }
377  return YES;
378 }
