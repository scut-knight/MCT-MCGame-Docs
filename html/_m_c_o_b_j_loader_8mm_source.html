46  faces.resize(m_faceCount * 9);
47 
48  // 注意，利用指针来修改vector的方法只适用于占用连续内存空间的vector类型。
49  // 而且一旦vector重新分配内存，指针会指向错误的地址。当然这里vector的内存已经事先分配好，
50  // 不过是在objFile大小可控的前提下，这种做法才是安全的。总之，我不推荐这种做法
51  float * v_ptr = &vertexs[0];
52  float * vn_ptr = &vertexNormals[0];
53  float * vt_ptr = &vertexTextures[0];
54  int * faces_ptr = &faces[0];
55  //Vertex* vertex = (Vertex*) &floats[0];
56  ifstream objFile([filename UTF8String]);
57  while (objFile) {
58  string c;
59  objFile >> c;
60  if (c == "v") {
61  // v vertex vertex vertex
62  objFile >> *(v_ptr++) >> *(v_ptr++) >> *(v_ptr++);
63  }
64  else{
65  if(c == "vn"){
66  // vn vertexNormal vertexNormal vertexNormal
67  objFile >> *(vn_ptr++) >> *(vn_ptr++) >> *(vn_ptr++);
68  }
69  else{
70  if(c == "vt"){
71  // vt vertexTexture vertexTexture vertexTexture
72  objFile >> *(vt_ptr++) >> *(vt_ptr++);
73  }
74  else{
75  if(c == "f"){
76  char d ;
77  // f face/face/face ...
78  objFile >> *(faces_ptr++) >> d >> *(faces_ptr++) >> d >> *(faces_ptr++);
79  objFile >> *(faces_ptr++) >> d >> *(faces_ptr++) >> d >> *(faces_ptr++);
80  objFile >> *(faces_ptr++) >> d >> *(faces_ptr++) >> d >> *(faces_ptr++);
81  }
82  }
83  }
84  }
85  // 忽略剩下的内容
86  objFile.ignore(MaxLineSize, '\n');
87  }
88  objFile.close();
