96 - (NSInteger)treeNodesApply:(MCTreeNode *)root withDeep:(NSInteger)deep{
97  switch (root.type) {
98  case ExpNode:
99  {
100  switch (root.value) {
101  case And:
102  return [self andNodeApply:root withDeep:deep];
103  case Or:
104  return [self orNodeApply:root withDeep:deep];
105  case Not:
106  return [self notNodeApply:root withDeep:deep];
107  default:
108  break;
109  }
110  }
111  break;
112 
113  case PatternNode:
114  {
115  switch (root.value) {
116  //if root.result == YES && deep == 0, just break and return YES
117  case Home:
118  {
119  if (root.result == YES && deep == 0) {
120  NSString *msg = [MCTransformUtil getContenFromPatternNode:root
121  accordingToWorkingMemory:self.workingMemory];
122  if (msg != nil) {
123  [self.accordanceMsgs addObject:msg];
124  }
125  }
126  }
127  break;
128 
129  case Check:
130  {
131  if (root.result == YES && deep == 0) {
132  NSString *msg = [MCTransformUtil getContenFromPatternNode:root
133  accordingToWorkingMemory:self.workingMemory];
134  if (msg != nil) {
135  [self.accordanceMsgs addObject:msg];
136  }
137  }
138  }
139  break;
140 
141  case CubiedBeLocked:
142  {
143  if (root.result == YES && deep == 0) {
144  NSString *msg = [MCTransformUtil getContenFromPatternNode:root
145  accordingToWorkingMemory:self.workingMemory];
146  if (msg != nil) {
147  [self.accordanceMsgs addObject:msg];
148  }
149  }
150  }
151  break;
152  // PatternNode
153  default:
154  return NO;
155  }
156  }
157  break;
158 
159  default:
160  return NO;
161  }
162  return YES;
163 }

170 - (BOOL)andNodeApply:(MCTreeNode *)root withDeep:(NSInteger)deep{
171  for (MCTreeNode *node in root.children) {
172 
173  //Being a 'and' node,
174  //it will be failed whenever one of its child is failed.
175  // 逻辑短路
176  if (![self treeNodesApply:node withDeep: deep+1]) {
177  return NO;
178  }
179 
180  //While this node is true, we store some accordance messages
181  //for several occasions.
182  switch (node.type) {
183  case ExpNode:
184  if (node.value == Not) {
185  //Notice that the node is 'Not' node.
186  //You should get the negative sentence by invoking
187  //'getNegativeSentenceOfContentFromPatternNode'
188  NSString *msg = [MCTransformUtil getNegativeSentenceOfContentFromPatternNode:[node.children objectAtIndex:0]
189  accordingToWorkingMemory:self.workingMemory];
190  if (msg != nil) {
191  [self.accordanceMsgs addObject:msg];
192  }
193  }
194  break;
195 
196  case PatternNode:
197  {
198  switch (node.value) {
199  case Home | Check:
200  {
201  NSString *msg = [MCTransformUtil getContenFromPatternNode:node
202  accordingToWorkingMemory:self.workingMemory];
203  if (msg != nil) {
204  [self.accordanceMsgs addObject:msg];
205  }
206  }
207  break;
208  case CubiedBeLocked:
209  {
210  // 有唯一子节点，并且这个子节点的值为0
211  if ([node.children count] == 0 ||
212  [(MCTreeNode *)[node.children objectAtIndex:0] value] == 0) {
213  NSString *msg = [MCTransformUtil getContenFromPatternNode:node
214  accordingToWorkingMemory:self.workingMemory];
