64 - (void)drawRect:(CGRect)rect {
65  CGContextRef context = UIGraphicsGetCurrentContext();
66  CGFloat locations[2] = { 0.0, 1.0 };
67  CGColorSpaceRef myColorspace = CGColorSpaceCreateDeviceRGB();
68  CGPoint start = CGPointMake(rect.size.width/2, rect.size.height/2);
69 
70  switch (gradientStyle) {
71  case UAGradientBackgroundStyleRadial: {
72  CGGradientRef myGradient = CGGradientCreateWithColorComponents(myColorspace, colorComponents, locations, 2);
73  double a = rect.size.width/2.0;
74  double b = rect.size.height/2.0;
75  double h = sqrt(a*a + b*b);
76  CGContextDrawRadialGradient(context, myGradient, start, 0, start, h, 0);
77  CGGradientRelease(myGradient);
78  break;
79  }
80  case UAGradientBackgroundStyleRadialReversed: {
81  CGFloat reversed[8] = { colorComponents[4], colorComponents[5], colorComponents[6], colorComponents[7], colorComponents[0], colorComponents[1], colorComponents[2], colorComponents[3] };
82  CGGradientRef myGradient = CGGradientCreateWithColorComponents (myColorspace, reversed, locations, 2);
83  double a = rect.size.width/2.0;
84  double b = rect.size.height/2.0;
85  double h = sqrt(a*a + b*b);
86  CGContextDrawRadialGradient(context, myGradient, start, 0, start, h, 0);
87  CGGradientRelease(myGradient);
88  break;
89  }
90  case UAGradientBackgroundStyleLinear: {
91  CGGradientRef myGradient = CGGradientCreateWithColorComponents (myColorspace, colorComponents, locations, 2);
92  CGContextDrawLinearGradient(context, myGradient, CGPointMake(0.0,0.0), CGPointMake(0.0,rect.size.height+1), 0);
93  CGGradientRelease(myGradient);
94  break;
95  }
96  case UAGradientBackgroundStyleLinearReversed: {
97  CGGradientRef myGradient = CGGradientCreateWithColorComponents (myColorspace, colorComponents, locations, 2);
98  CGContextDrawLinearGradient(context, myGradient, CGPointMake(0.0,rect.size.height+1), CGPointMake(0.0,0.0), 0);
99  CGGradientRelease(myGradient);
100  break;
101  }
102  case UAGradientBackgroundStyleCenterHighlight: {
103  CGFloat myLocations[3] = { 0.0, 0.5, 1.0 };
104  CGFloat myColorComponents[12] = { colorComponents[0], colorComponents[1], colorComponents[2], colorComponents[3], colorComponents[4], colorComponents[5], colorComponents[6], colorComponents[7], colorComponents[0], colorComponents[1], colorComponents[2], colorComponents[3] };
105  CGGradientRef myGradient = CGGradientCreateWithColorComponents (myColorspace, myColorComponents, myLocations, 3);
106  CGContextDrawLinearGradient(context, myGradient, CGPointMake(0.0,0.0), CGPointMake(0.0,rect.size.height+1), 0);
107  CGGradientRelease(myGradient);
108  break;
109  }
110  default:
111  break;
112  }
113  CGColorSpaceRelease(myColorspace);
114 
115  if (lineMode & UAGradientLineModeTop ||
116  lineMode & UAGradientLineModeTopAndBottom) {
117  // top Line
118  CGContextSetRGBStrokeColor(context, 1, 1, 1, 1.0);
119  CGContextMoveToPoint(context, 0, 0);
120  CGContextAddLineToPoint(context, self.frame.size.width, 0);
121  CGContextStrokePath(context);
122  }

128  CGContextAddLineToPoint(context, self.frame.size.width, self.frame.size.height);
129  CGContextStrokePath(context);
